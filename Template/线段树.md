# 线段树

## 1. 动态开点 - 估点

```c++
#include <bits/stdc++.h>

using namespace std;

struct Node
{
    int l, r;
    int val;
    // lazy 标记
    int lazy;
} tr[10000];

class Solution
{
public:
    // 从左右儿子更新父节点
    void pushup(int u)
    {
        tr[u].val = max(tr[u << 1].val, tr[u << 1 | 1].val);
    }

    // 懒惰lazy标记下方
    void pushdown(int u)
    {
        if (tr[u].lazy)
        {
            tr[u << 1].lazy = 1;
            tr[u << 1 | 1].lazy = 1;
            tr[u << 1].val = tr[u << 1 | 1].val = tr[u].val;
            tr[u].lazy = 0;
        }
    }
    void build(int u, int l, int r)
    {
        if (l == r)
            tr[u] = {l, r, 0, 0};
        else
        {
            tr[u] = {l, r, 0, 0};
            int mid = (l + r) >> 1;
            build(u << 1, l, mid);
            build(u << 1 | 1, mid + 1, r);
            pushup(u);
        }
    }
    void update(int u, int l, int r, int v)
    {
        if (tr[u].l >= l && tr[u].r <= r)
        {
            tr[u].lazy = 1;
            tr[u].val = v;
        }
        else
        {
            pushdown(u);
            int mid = (tr[u].l + tr[u].r) >> 1;
            if (l <= mid)
                update(u << 1, l, r, v);
            if (r > mid)
                update(u << 1 | 1, l, r, v);
            pushup(u);
        }
    }
    // 区间更新为val
    void update(int cur, int s, int t, int l, int r, int val)
    {
        // 目前在 tr[cur], [s, t] 为 tr[cur]左右区间边界, 搜索区间为 [l, r]
        // 修改区间[l, r]的值为 val

        // 当前区间 为 询问区间 的子集时, 直接修改区间的val
        // 并更新 lazy 懒惰标记
        if (l <= s && t <= r)
        {
            tr[cur].val = val;
            tr[cur].lazy = val;
            return;
        }
    }
    int query(int u, int l, int r)
    {
        if (tr[u].l >= l && tr[u].r <= r)
            return tr[u].val;
        pushdown(u);
        int mid = (tr[u].l + tr[u].r) >> 1;
        int res = 0;
        if (l <= mid)
            res = max(res, query(u << 1, l, r));
        if (r > mid)
            res = max(res, query(u << 1 | 1, l, r));
        return res;
    }

    int query(int cur, int s, int t, int l, int r)
    {
        // 目前在 tr[cur], [s, t] 为 tr[cur]左右区间边界, 搜索区间为 [l, r]

        // 当前区间 为 询问区间 的子集时, 直接返回当前区间的值
        if (l <= s && t <= r)
        {
            return st[u].val;
        }
        // 
        int mid = s + ((t - s) >> 1);
        int ans = 0;

        // 如果左儿子代表的区间 [s, mid] 与询问区间有交集, 则递归查询左儿子
        if (l <= mid) ans = max(ans, query(mid, s, mid, l, r));
        // 如果右儿子代表的区间 [mid + 1, t] 与询问区间有交集, 则递归查询右儿子
        if (mid <= r) ans = max(ans, query(mid, mid + 1, t, l, r));

        return ans;
    }

    // 本题
    vector<int> fallingSquares(vector<vector<int>> &positions)
    {
        /* 
            离散化
        */
        set<int> arr;
        for (auto p : positions)
        {
            arr.insert(p[0]);
            arr.insert(p[0] + p[1] - 1);
        }
        int n = 0;
        unordered_map<int, int> S;
        for (auto p : arr)
            S[p] = ++n;

        // 构建线段树[1, n]
        build(1, 1, n);

        vector<int> res(positions.size());
        for (int i = 0; i < positions.size(); i++)
        {
            /*
                不同方块之间的边缘可以重合, 但不会导致方块叠加
                因此我们当我们对一个区间 [a, b] 进行操作（查询或插入）时, 可以将其调整为 [a, b - 1]
            */

            int l = S[positions[i][0]];
            int r = S[positions[i][0] + positions[i][1] - 1];

            // 1. 查询当前范围 [a, b] 的最大高度为多少, 假设为 cur
            int cur = query(1, l, r);

            // 2. 更新当前范围 [a, b] 的最新高度为 cur + len
            update(1, l, r, cur + positions[i][1]);

            // 
            res[i] = tr[1].val;
        }
        return res;
    }
};
```

```java
class Solution {
    class Node {
        // ls 和 rs 分别代表当前区间的左右子节点所在 tr 数组中的下标
        // val 代表当前区间的最大高度, lazy 为懒标记
        int ls, rs, val, lazy;
    }
    int N = (int)1e9, cnt = 0;
    Node[] tr = new Node[1000010];
    void update(int u, int lc, int rc, int l, int r, int v) {
        if (l <= lc && rc <= r) {
            tr[u].val = v;
            tr[u].lazy = v;
            return ;
        }
        pushdown(u);
        int mid = lc + rc >> 1;
        if (l <= mid) update(tr[u].ls, lc, mid, l, r, v);
        if (r > mid) update(tr[u].rs, mid + 1, rc, l, r, v);
        pushup(u);
    }
    int query(int u, int lc, int rc, int l, int r) {
        if (l <= lc && rc <= r) return tr[u].val;
        pushdown(u);
        int mid = lc + rc >> 1, ans = 0;
        if (l <= mid) ans = query(tr[u].ls, lc, mid, l, r);
        if (r > mid) ans = Math.max(ans, query(tr[u].rs, mid + 1, rc, l, r));
        return ans;
    }
    void pushdown(int u) {
        if (tr[u] == null) tr[u] = new Node();
        if (tr[u].ls == 0) {
            tr[u].ls = ++cnt;
            tr[tr[u].ls] = new Node();
        }
        if (tr[u].rs == 0) {
            tr[u].rs = ++cnt;
            tr[tr[u].rs] = new Node();
        }
        if (tr[u].lazy == 0) return ;
        int lazy = tr[u].lazy;
        tr[tr[u].ls].lazy = lazy; tr[tr[u].rs].lazy = lazy;
        tr[tr[u].ls].val = lazy; tr[tr[u].rs].val = lazy;
        tr[u].lazy = 0;
    }
    void pushup(int u) {
        tr[u].val = Math.max(tr[tr[u].ls].val, tr[tr[u].rs].val);
    }
    public List<Integer> fallingSquares(int[][] ps) {
        List<Integer> ans = new ArrayList<>();
        tr[1] = new Node();
        for (int[] info : ps) {
            int x = info[0], h = info[1], cur = query(1, 1, N, x, x + h - 1);
            update(1, 1, N, x, x + h - 1, cur + h);
            ans.lazy(tr[1].val);
        }
        return ans;
    }
}
```
